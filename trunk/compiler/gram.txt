Початковий символ граматики – <файл>.
<Файл>                            ::= <Публ. клас>
 | < Публ. класс> < Послід. непубл. класів>  
<Послід. непубл. класів>  ::= <Непубл. клас>
 | < Непубл. клас> < Послід. непубл. классов>  
<Публ. клас>                  ::= public class <Ідентифікатор> { <Тіло класу>}
<Непубл. клас>              ::= class <Ідентифікатор> { <Тіло класу>}
<Тіло класу>                    ::= <Елем. тіла класу> 
|  <Елем. тіла класу> <Тіло класу>
<Елем. тіла класу>	::=  	<Опис функції> 
                                          		|<Опис змінної> 
                                          		|<Опис константи>	
<Опис функції>                ::= <Оголошення функції> <Тіло функції>
<Опис змінної>                 ::= <Тип змінної> <Назва змінної>;
<Опис константи>            ::= final <Тип змінної> <Назва змінної> = <Літерал>;
<Літерал>                         ::= <Строковий літерал> |
				      <Булевий літерал> | 
				      <Числовий літерал>
<Строковий літерал>         ::= “<Послiдовнiсть символiв>”
<Послiдовнiсть символiв>::= <Символ> 
|  <Cимвол><Послiдовнiсть символiв>
| <Пусто>

<Булевий літерал>           ::= true | false
<Числовий літерал>         ::= <Ціле> | 
                                                .<Ціле невідємне>|
                                                <Ціле>.<Ціле невідємне>
<Ціле>                             ::= <Ціле невідємне>| +<Ціле невідємне>| -<Ціле невідємне>
<Ціле невідємне>            ::= <Цифра> | <Цифра><Ціле невідємне>
<Назва змінної>               ::= <Ідентифікатор>
<Ідентифікатор>              ::= <Літера>|
                                                <Літера><Ідентифікатор>
<Оголошення функції>     ::= public <Тип результату> <Назва функції> (<Список формальних параметрів>)
<Тип результату>             ::= <Тип змінної> | void
<Тип змiнної>                 ::= <Тип> | <Тип>[]
<Тип>                               ::= boolean | int | String | float |<Назва типу>
<Назва типу>                   ::= <Ідентифікатор>
<Назва функції>               ::= <Ідентифікатор>
<Літера>                           ::= A|..|Z|a|..|z|_
<Цифра>                           ::= 0|..|9
<Список форм. парам.>    ::= < Опис змінної> 
                                             | <Опис константи>
                                  | <Опис змінної>, <Список формальних параметрів>

<Змінна>                          ::= <Ідентифікатор>
<Назва методу>               ::= <Ідентифікатор>
<Константа>                     ::= <Ідентифікатор>
<Тіло функції>                  ::= {<Блок> return <Вираз>;} |{<Блок> return;}
<Блок>                              ::= <Оператор>; | <Оператор>; <Блок>
<Оператор>                       ::= <Вираз> |
                                                 <Змінна>=<Вираз> |
                                                 <Умовний оператор> |
                                                 <Оператор циклу>
<Умовний оператор>        ::=if (<Вираз>) {<Блок>} |
                                               if (<Вираз>) {<Блок>} else{<Блок>} 
<Вираз>                           ::= <ОперандАБО> | <ОперандАБО> || <Вираз> 
<ОперандАБО>                 ::= <ОперандІ> | <ОперандІ> && <ОперандАБО> 
<ОперандІ>                      ::= <Вираз порівняння> 
| <Вираз порівняння> <Операція рівності> <ОперандI> 

<Вираз порівняння>         ::= <Вираз порівняння I>
 | <Вираз порівняння I> <Операція рівності> <Вираз порiвняння>

<Вираз порiвняння I> ::= <Простий вираз> 
|  <Простий вираз> <Операцiя порiвняння> <Вираз порiвняння I>

<Простий вираз>             ::= <Доданок>
 | <Доданок> <Аддитивна операція> <Простий вираз>

<Доданок>                       ::= <Множник> 
| <Множник> <Мультиплікативна операція> <Доданок>

<Множник>                      ::= <Вираз приведення> 
| ! <Вираз приведення>


<Вираз приведення>      ::= <Елементарний вираз> 
| (<Тип>)<Елементарний вираз >

< Елементарний вираз >    ::= <Змінна> |
                                                <Константа> |
                                                <Літерал> |
                                                (<Вираз >) |
                                                <Посилання>.<Назва функції> (<Список фактичних параметрів>)
                                                <Посилання>.<Назва змінної>

<Список фактичних параметр>    ::= <Вираз> | <Вираз>, <Список фактичних параметрів>

<Посилання>                          ::= <Змінна> |
                                                (<Посилання>) |
                                                <Посилання>.<Ідентифікатор> |
                                                <Посилання>[<Вираз>]


<Операція порівняння>     ::= > | < |  >= | <=|
<Операція рівності>          ::= == | !=
<Аддитивна операція>      ::= + | -
<Мультиплікативна>         ::= * | / | %
